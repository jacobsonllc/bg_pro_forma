// Process uploaded file async function processFile(file) { if (!file.name.endsWith('.json')) { showError('Please upload a JSON file.'); return; } showLoading(); try { const text = await file.text(); const jsonData = JSON.parse(text); // Handle both direct JSON and array format from N8N let processedData; if (Array.isArray(jsonData) && jsonData.length > 0) { processedData = jsonData[0]; // N8N wraps data in array } else { processedData = jsonData; } // Validate required data structure if (!processedData.monthlyData || !processedData.periodSummaries) { throw new Error('Invalid data structure. Missing monthlyData or periodSummaries.'); } dashboardData = processedData; hideLoading(); showDashboard(); // Add small delay to ensure DOM is ready setTimeout(() => { updateDashboard(); }, 100); } catch (error) { hideLoading(); showError('Error reading file: ' + error.message); console.error('File processing error:', error); } }
